PostgreSql获取过去或未来的某个时间点

	select now() + interval '2 years';//二年后的时间
	
	select now() + interval '2 years';//二年后的时间
	
	select now() + interval '2 year';//二年后的时间
	
	select now() + interval '2 y';//二年后的时间
	
	select now() + interval '2 Y';//二年后的时间
	
	select now() + interval '2Y';//二年后的时间
	
	select now() + interval '1 month';//1月后
	
	select now() - interval '-3 week';//三周前
	
	select now() + '10 min';//10分钟后
